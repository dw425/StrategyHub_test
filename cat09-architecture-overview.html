Enterprise Architecture Advisory

Analyzes organizational needs, evaluates architecture patterns, recommends decomposition strategies, and generates implementation roadmaps for modern distributed systems.

  • Pattern selection based on requirements
  • Domain boundary identification
  • Service decomposition planning
  • Event storming facilitation
  • Data ownership mapping
  • Migration strategy generation
  • Anti-pattern detection
Python
YAML
architecture_crew.py
# Architecture Advisory Crew
from crewai import Agent, Task, Crew, Process
from langchain_anthropic import ChatAnthropic

# Initialize LLM
llm = ChatAnthropic(model="claude-sonnet-4-20250514")

# Define Architecture Agents
domain_analyst = Agent(
    role="Domain Analyst",
    goal="Identify bounded contexts and domain boundaries",
    backstory="""Expert in Domain-Driven Design, specializing
    in strategic design and context mapping.""",
    llm=llm,
    tools=[domain_modeler, context_mapper, event_stormer]
)

pattern_advisor = Agent(
    role="Pattern Advisor",
    goal="Recommend architecture patterns for requirements",
    backstory="""Architect with deep knowledge of microservices,
    event-driven, data mesh, and lakehouse patterns.""",
    llm=llm,
    tools=[pattern_matcher, trade_off_analyzer, decision_matrix]
)

migration_planner = Agent(
    role="Migration Planner",
    goal="Create incremental migration roadmaps",
    backstory="""Specialist in strangler fig pattern and 
    legacy modernization strategies.""",
    llm=llm,
    tools=[dependency_analyzer, risk_assessor, roadmap_generator]
)

# Create Architecture Crew
architecture_crew = Crew(
    agents=[domain_analyst, pattern_advisor, migration_planner],
    process=Process.sequential,
    verbose=True
)

async def analyze_architecture(system_description):
    """Run architecture analysis and recommendations."""
    tasks = [
        Task(description="Map domain boundaries", agent=domain_analyst),
        Task(description="Recommend patterns", agent=pattern_advisor),
        Task(description="Create migration plan", agent=migration_planner)
    ]
    return await architecture_crew.kickoff_async(tasks=tasks)
